// --- NUEVA LÓGICA PARA GESTIÓN DE VENTAS Y DEVOLUCIONES --- function renderLocales() { const cont = document.getElementById('tab-locales'); if (permitidos.length === 0) { cont.innerHTML = `

No hay productos en locales actualmente.

`; return; } cont.innerHTML = permitidos.map(p => `
${p.local} REF: ${p.ref || 'S/R'}
${p.name}
Talla: ${p.size}
`).join(''); } // Acción 1: El producto se vendió (Solo se borra del local) async function registrarVenta(id) { if(confirm("¿Confirmar venta? Se quitará de la lista del local.")) { await db.collection("permitidos").doc(id).delete(); // El sistema se actualiza solo por el onSnapshot } } // Acción 2: El producto regresa a la bodega (Suma +1 al inventario) async function devolverABodega(id, nombre, talla) { if(confirm(`¿Reingresar ${nombre} (Talla ${talla}) a la bodega?`)) { try { // 1. Buscar el producto original en el inventario por nombre o referencia const snap = await db.collection("inventory").where("name", "==", nombre).get(); if(!snap.empty) { const prodDoc = snap.docs[0]; const pData = prodDoc.data(); const nuevasTallas = {...pData.sizes}; // Sumar 1 a la talla correspondiente nuevasTallas[talla] = (nuevasTallas[talla] || 0) + 1; // 2. Actualizar bodega await db.collection("inventory").doc(prodDoc.id).update({ sizes: nuevasTallas }); // 3. Borrar de la lista de locales await db.collection("permitidos").doc(id).delete(); alert("Producto reingresado a bodega con éxito."); } else { alert("Error: No se encontró el producto original en bodega para devolverlo."); } } catch (e) { console.error(e); alert("Error al procesar la devolución."); } } }
